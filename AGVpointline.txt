#创建point队列 
points=[]

#添加点方法 
def point_add(x,y):
    a=[x,y]
    points.append(a)
    
#创建点 
point_add(1,2)
point_add(1,3)
point_add(1,4)
point_add(2,4)

print(points)


#创建line队列 
lines=[]

#添加线的方法 
def line_add(x1,y1,x2,y2):
    b=[x1,y1,x2,y2]
    lines.append(b)

#创建线 
line_add(1,2,1,3)
line_add(1,3,1,4)
line_add(1,4,2,4)

print(lines)

    for point in points:
        if point == [1,2]:
